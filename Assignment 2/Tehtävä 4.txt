// Käytin yksinkertaista for silmukkaa matriisin läpikäymisessä
// Esimmäinen rivi on [0][0-2], seuraava rivi [1][0-2] ja viimeinen [2][0-2]
// Jos taulukon arvo vastaa käyttäjän syöttämää arvoa, tulostetaan arvon paikka ja 
// palautetaan taulukon arvo muuttujaan m
// Pääohjelmassa verrataan käyttäjän syöttämää lukua ja palautettua arvoa ja saadaan ulos
// toinen syöte, joka on tässä tapauksessa turha, koska todennus tapahtuu jo if lausekkeessa
// Jos mikään arvo ei toteudu niin palautetaan 0, joka vertaa käyttäjän syötteeseen(Käyttäjän syöte ei voi olla 0).
// Arvot eivät päde joten tulosteaan syöte, missä kerrotaan että arvoa ei löytynyt matriisista.

#include <iostream>

using namespace std;
int taulukko[3][3] = {
	{ 10, 11, 232 },
	{ 407, 15, 62 },
	{ 18, 29, 100 }
};

int matriisi_haku(int n) {
	for (int i = 0; i < 3; i++) {
		for (int k = 0; k < 3; k++) {
			if (taulukko[i][k] == n) {
				cout << "Value is in " << i << " " << k << endl;
				return taulukko[i][k];
			}
		}
	}
	return 0;
}

int main() {
	int n = 0;
		cout << "Search a value inside the 2d array.\nPlease input itegers only\n> ";
	while (n == 0) {
		cin >> n;
	}
	int m = matriisi_haku(n);
	if (n == m) { cout << n << " is inside the 2d array\n"; }
	else { cout << "Unfortunately " << n << " is not inside the 2d array\n"; }
}
